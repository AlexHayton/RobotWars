Check for command line arguments

var debugMode = false;
if (process.argv.length > 0) {
  process.argv.forEach(function(val, index) {
      if (val == "--help")
      {
        console.log("Instructions for use:");
        process.exit();
      }
      
      if (val == "--debug")
      {
        console.log("Debug mode enabled");
        debugMode = true;
      }
  });
}
